349 nlohmann::json http_connection::get_json_row(const CassResult* result,
350  CassIterator* rows) const {
351  size_t column_count = cass_result_column_count(result);
352  const CassRow* row = cass_iterator_get_row(rows);
353  nlohmann::json json_row;
354 
355  for (size_t i = 0; i < column_count; ++i) {
356  const char* column_name;
357  size_t column_name_length;
358  cass_result_column_name(result, i, &column_name, &column_name_length);
359  std::string col_name(column_name, column_name_length);
360 
361  const CassValue* column = cass_row_get_column(row, i);
362 
363  CassValueType type = cass_value_type(column);
364 
365  switch (type) {
366  case CASS_VALUE_TYPE_UUID: {
367  CassUuid uuid;
368  char uuid_str[CASS_UUID_STRING_LENGTH];
369  cass_value_get_uuid(column, &uuid);
370  cass_uuid_string(uuid, uuid_str);
371  json_row[col_name] = uuid_str;
372  break;
373  }
374  case CASS_VALUE_TYPE_BIGINT: {
375  cass_int64_t column_value;
376  cass_value_get_int64(column, &column_value);
377  json_row[col_name] = column_value;
378  break;
379  }
380  case CASS_VALUE_TYPE_BOOLEAN: {
381  cass_bool_t column_value;
382  cass_value_get_bool(column, &column_value);
383  json_row[col_name] = (column_value == cass_true ? true : false);
384  break;
385  }
386  default: {
387  const char* column_value;
388  size_t column_value_length;
389  cass_value_get_string(column, &column_value, &column_value_length);
390  std::string col_value(column_value, column_value_length);
391  json_row[col_name] = col_value;
392  break;
393  }
394  }
395  }
396 
397  return json_row;
398 }
