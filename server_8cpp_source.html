319 void http_connection::handle_query_result(CassFuture* result_future) {
320  const CassResult* result = cass_future_get_result(result_future);
321  auto rows = std::unique_ptr<CassIterator,
322  decltype(&cass_iterator_free)>(cass_iterator_from_result(result), &cass_iterator_free);
323  nlohmann::json json_array = nlohmann::json::array();
324  size_t total_rows = cass_result_row_count(result);
325 
326  long long page = std::any_cast<long long>(request_un_map_["pagination-page"]);
327  int per_page = std::any_cast<int>(request_un_map_["pagination-per-page"]);
328 
329  size_t start_row = per_page * (page - 1);
330  size_t end_row = per_page * page;
331 
332  for (size_t i = 0; cass_iterator_next(rows.get()) && i < end_row; ++i) {
333  if (i >= start_row) {
334  json_array.push_back(get_json_row(result, rows.get()));
335  }
336  }
337 
338  auto total_page_count =
339  static_cast<long long>(std::ceil(static_cast<double>(total_rows) / per_page));
340  response_.set("Pagination-Total-Pages", std::to_string(total_page_count));
341  response_.set("Pagination-Total-Comments", std::to_string(total_rows));
342 
343  cass_result_free(result);
344  beast::ostream(response_.body()) << json_array;

347 nlohmann::json http_connection::get_json_row(const CassResult* result,
348  CassIterator* rows) const {
349  size_t column_count = cass_result_column_count(result);
350  const CassRow* row = cass_iterator_get_row(rows);
351  nlohmann::json json_row;
352 
353  for (size_t i = 0; i < column_count; ++i) {
354  const char* column_name;
355  size_t column_name_length;
356  cass_result_column_name(result, i, &column_name, &column_name_length);
357  std::string col_name(column_name, column_name_length);
358 
359  const CassValue* column = cass_row_get_column(row, i);
360 
361  CassValueType type = cass_value_type(column);
362 
363  switch (type) {
364  case CASS_VALUE_TYPE_UUID: {
365  CassUuid uuid;
366  char uuid_str[CASS_UUID_STRING_LENGTH];
367  cass_value_get_uuid(column, &uuid);
368  cass_uuid_string(uuid, uuid_str);
369  json_row[col_name] = uuid_str;
370  break;
371  }
372  case CASS_VALUE_TYPE_BIGINT: {
373  cass_int64_t column_value;
374  cass_value_get_int64(column, &column_value);
375  json_row[col_name] = column_value;
376  break;
377  }
378  case CASS_VALUE_TYPE_BOOLEAN: {
379  cass_bool_t column_value;
380  cass_value_get_bool(column, &column_value);
381  json_row[col_name] = (column_value == cass_true ? true : false);
382  break;
383  }
384  default: {
385  const char* column_value;
386  size_t column_value_length;
387  cass_value_get_string(column, &column_value, &column_value_length);
388  std::string col_value(column_value, column_value_length);
389  json_row[col_name] = col_value;
390  break;
391  }
392  }
393  }
394 
395  return json_row;
396 }
